public _NWD

.code
_NWD PROC ; int NWD(UCHAR a, UCHAR b);
	push ebp
	mov ebp, esp
	mov eax, [ebp + 8] ; a
	mov edx, [ebp + 12] ; b
	cmp eax, edx
	je koniec
	ja a_wieksze_od_b; a>b

; else b>a
	sub edx, eax
	push edx
	push eax
	call _NWD
	add esp, 8
	jmp koniec

a_wieksze_od_b:
	sub eax, edx
	push edx
	push eax
	call _NWD
	add esp, 8

koniec:
	pop ebp
	ret
_NWD ENDP

END
